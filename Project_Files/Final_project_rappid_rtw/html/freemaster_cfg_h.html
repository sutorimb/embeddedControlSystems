11#ifndef __FREEMASTER_CFG_H



12#define __FREEMASTER_CFG_H

14/******************************************************************************



15 * Select interrupt or poll-driven serial communication



16 ******************************************************************************/



17#define FMSTR_LONG_INTR                0                         /* complete message processing in interrupt */



18#define FMSTR_SHORT_INTR               0                         /* SCI FIFO-queuing done in interrupt */



19#define FMSTR_POLL_DRIVEN              1                         /* no interrupt needed, polling only */



20



21/*****************************************************************************



22 * Select communication interface (SCI base address)



23 ******************************************************************************/



24#include "freemaster_device.h"



25#ifndef FMSTR_SCI_SECOND



26#define FMSTR_SCI_BASE                 0xFFE40000UL              /* LINFlex0 base on MPC5643L */



27#else



28#define FMSTR_SCI_BASE                 0xFFE44000UL              /* LINFlex1 base on MPC5643L */



29#endif



30



31#ifndef FMSTR_CAN_SECOND



32#define FMSTR_CAN_BASE                 0xFFFC0000UL              /* FlexCAN0 base on MPC5643L */



33#else



34#define FMSTR_CAN_BASE                 0xFFFC4000UL              /* FlexCAN1 base on MPC5643L */



35#endif



36



37#define FMSTR_USE_SCI                  1



38#define FMSTR_USE_FLEXCAN              0



39



40/* select RX and TX FlexCAN Message buffers */



41#define FMSTR_FLEXCAN_TXMB             0



42#define FMSTR_FLEXCAN_RXMB             1



43



44/******************************************************************************



45 * Input/output communication buffer size



46 ******************************************************************************/



47#define FMSTR_COMM_BUFFER_SIZE         0                         /* set to 0 for "automatic" */



48



49/******************************************************************************



50 * Receive FIFO queue size (use with FMSTR_SHORT_INTR only)



51 ******************************************************************************/



52#define FMSTR_COMM_RQUEUE_SIZE         32                        /* set to 0 for "default" */



53



54/*****************************************************************************



55 * Support for Application Commands



56 ******************************************************************************/



57#define FMSTR_USE_APPCMD               1                         /* enable/disable App.Commands support */



58#define FMSTR_APPCMD_BUFF_SIZE         32                        /* App.Command data buffer size */



59#define FMSTR_MAX_APPCMD_CALLS         4                         /* how many app.cmd callbacks? (0=disable) */



60



61/*****************************************************************************



62 * Oscilloscope support



63 ******************************************************************************/



64#define FMSTR_USE_SCOPE                1                         /* enable/disable scope support */



65#define FMSTR_MAX_SCOPE_VARS           8                         /* max. number of scope variables (2..8) */



66



67/*****************************************************************************



68 * Recorder support



69 ******************************************************************************/



70#define FMSTR_USE_RECORDER             1                         /* enable/disable recorder support */



71#define FMSTR_MAX_REC_VARS             8                         /* max. number of recorder variables (2..8) */



72#define FMSTR_REC_OWNBUFF              0                         /* use user-allocated rec. buffer (1=yes) */



73



74/* built-in recorder buffer (use when FMSTR_REC_OWNBUFF is 0) */



75#define FMSTR_REC_BUFF_SIZE            1024                      /* built-in buffer size */



76



77/* recorder time base, specifies how often the recorder is called in the user app. */



78#define FMSTR_REC_TIMEBASE             FMSTR_REC_BASE_MILLISEC(0) /* 0 = "unknown" */



79#define FMSTR_REC_FLOAT_TRIG           0                         /* enable/disable floating point triggering */



80



81/*****************************************************************************



82 * Target-side address translation (TSA)



83 ******************************************************************************/



84#define FMSTR_USE_TSA                  0                         /* enable TSA functionality */



85#define FMSTR_USE_TSA_SAFETY           0                         /* enable access to TSA variables only */



86#define FMSTR_USE_TSA_INROM            0                         /* TSA tables declared as const (put to ROM) */



87



88/*****************************************************************************



89 * Enable/Disable read/write memory commands



90 ******************************************************************************/



91#define FMSTR_USE_READMEM              1                         /* enable read memory commands */



92#define FMSTR_USE_WRITEMEM             1                         /* enable write memory commands */



93#define FMSTR_USE_WRITEMEMMASK         1                         /* enable write memory bits commands */



94



95/*****************************************************************************



96 * Enable/Disable read/write variable commands (a bit faster than Read Mem)



97 ******************************************************************************/



98#define FMSTR_USE_READVAR              0                         /* enable read variable fast commands */



99#define FMSTR_USE_WRITEVAR             0                         /* enable write variable fast commands */



100#define FMSTR_USE_WRITEVARMASK         0                         /* enable write variable bits fast commands */



101#endif                                 /* __FREEMASTER_CFG_H */
