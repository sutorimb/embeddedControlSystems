24/*-------------------------------------



25   command message - standard commands



26   -------------------------------------*/



27#define FMSTR_CMD_READMEM              0x01U



28#define FMSTR_CMD_WRITEMEM             0x02U



29#define FMSTR_CMD_WRITEMEMMASK         0x03U



30#define FMSTR_CMD_READMEM_EX           0x04U                     /* read a block of memory */



31#define FMSTR_CMD_WRITEMEM_EX          0x05U                     /* write a block of memory */



32#define FMSTR_CMD_WRITEMEMMASK_EX      0x06U                     /* write block of memory with bit mask */



33#define FMSTR_CMD_SETUPSCOPE           0x08U



34#define FMSTR_CMD_SETUPREC             0x09U



35#define FMSTR_CMD_SETUPSCOPE_EX        0x0aU                     /* setup the oscilloscope */



36#define FMSTR_CMD_SETUPREC_EX          0x0bU                     /* setup the recorder */



37#define FMSTR_CMD_SENDAPPCMD           0x10U                     /* send the application command */



38#define FMSTR_CMD_GETTSAINFO           0x11U                     /* get TSA info */



39#define FMSTR_CMD_GETTSAINFO_EX        0x12U                     /* get TSA info 32bit */



40#define FMSTR_CMD_SFIOFRAME_1          0x13U                     /* deliver & execute SFIO frame (even) */



41#define FMSTR_CMD_SFIOFRAME_0          0x14U                     /* deliver & execute SFIO frame (odd) */



42#define FMSTR_CMD_PIPE                 0x15U                     /* read/write pipe data */

44/*-------------------------------------



45   command message - Fast Commands



46   -------------------------------------*/



47



48/* no data part */



49#define FMSTR_CMD_GETINFO              0xc0U                     /* retrieve board information structure */



50#define FMSTR_CMD_STARTREC             0xc1U                     /* start data recorder */



51#define FMSTR_CMD_STOPREC              0xc2U                     /* stop data recorder */



52#define FMSTR_CMD_GETRECSTS            0xc3U                     /* get the recorder status */



53#define FMSTR_CMD_GETRECBUFF           0xc4U



54#define FMSTR_CMD_READSCOPE            0xc5U                     /* read the scope data */



55#define FMSTR_CMD_GETAPPCMDSTS         0xc6U                     /* get the application command status */



56#define FMSTR_CMD_GETINFOBRIEF         0xc8U                     /* retrieve a subset of board information structure */



57#define FMSTR_CMD_GETRECBUFF_EX        0xc9U                     /* get the recorder data */



58#define FMSTR_CMD_SFIOGETRESP_0        0xcaU                     /* retry to get last SFIO response (even) */



59#define FMSTR_CMD_SFIOGETRESP_1        0xcbU                     /* retry to get last SFIO response (odd) */



60



61/* 2 bytes data part */



62#define FMSTR_CMD_READVAR8             0xD0U



63#define FMSTR_CMD_READVAR16            0xD1U



64#define FMSTR_CMD_READVAR32            0xD2U



65#define FMSTR_CMD_GETAPPCMDDATA        0xD3U                     /* get the application command data */



66#define FMSTR_CMD_GETSTRLEN            0xD4U                     /* get string length (required by TSA) */



67



68/* 4 bytes data part */



69#define FMSTR_CMD_READVAR8_EX          0xe0U                     /* read byte variable */



70#define FMSTR_CMD_READVAR16_EX         0xe1U                     /* read word variable */



71#define FMSTR_CMD_READVAR32_EX         0xe2U                     /* read dword variable */



72#define FMSTR_CMD_WRITEVAR8            0xe3U                     /* write byte variable */



73#define FMSTR_CMD_WRITEVAR16           0xe4U                     /* write word variable */



74#define FMSTR_CMD_WRITEVAR8MASK        0xe5U                     /* write specified bits in byte variable  */



75#define FMSTR_CMD_GETSTRLEN_EX         0xe6U                     /* get string length (required by TSA) */



76



77/* 6 bytes data part */



78#define FMSTR_CMD_WRITEVAR32           0xf0U                     /* write dword variable */



79#define FMSTR_CMD_WRITEVAR16MASK       0xf1U                     /* write specified bits in word variable */



80

84



85/* flags in response codes */



86#define FMSTR_STSF_ERROR               0x80U                     /* FLAG: error answer (no response data) */



87#define FMSTR_STSF_VARLEN              0x40U                     /* FLAG: variable-length answer (length byte) */



88#define FMSTR_STSF_EVENT               0x20U                     /* FLAG: reserved */



89



90/* confirmation codes */



91#define FMSTR_STS_OK                   0x00U                     /* operation finished successfully */



92#define FMSTR_STS_RECRUN               0x01U                     /* data recorder is running */



93#define FMSTR_STS_RECDONE              0x02U                     /* data recorder is stopped */



94



95/* error codes */



96#define FMSTR_STC_INVCMD               0x81U                     /* unknown command code */



97#define FMSTR_STC_CMDCSERR             0x82U                     /* command checksum error */



98#define FMSTR_STC_CMDTOOLONG           0x83U                     /* command is too long */



99#define FMSTR_STC_RSPBUFFOVF           0x84U                     /* the response would not fit into transmit buffer */



100#define FMSTR_STC_INVBUFF              0x85U                     /* invalid buffer length or operation */



101#define FMSTR_STC_INVSIZE              0x86U                     /* invalid size specified */



102#define FMSTR_STC_SERVBUSY             0x87U                     /* service is busy */



103#define FMSTR_STC_NOTINIT              0x88U                     /* service is not initialized */



104#define FMSTR_STC_EACCESS              0x89U                     /* access is denied */



105#define FMSTR_STC_SFIOERR              0x8AU                     /* Error in SFIO frame */



106#define FMSTR_STC_SFIOUNMATCH          0x8BU                     /* Even/odd mismatch in SFIO transaction */



107#define FMSTR_STC_PIPEERR              0x8CU                     /* Pipe error */



108#define FMSTR_STC_FASTRECERR           0x8DU                     /* Feature not implemented in Fast Recorder */



109#define FMSTR_STC_CANTGLERR            0x8EU                     /* CAN fragmentation (toggle bit) error */



110#define FMSTR_STC_CANMSGERR            0x8FU                     /* CAN message format error */



111#define FMSTR_STC_FLOATDISABLED        0x90U                     /* Floating point triggering is not enabled */
