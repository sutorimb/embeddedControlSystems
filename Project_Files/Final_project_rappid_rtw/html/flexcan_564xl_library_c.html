37  /******************************************************************************



38   *   Function:  flexcan_564xl_init



39     @brief    Initialize FlexCAN module .



40     @details   Initialize FlexCAN module



41



42     @param[in]     flex_can     FlexCAN module



43     @param[in]     MCR     	MCR



44     @param[in]     CR     	CR



45



46     @return status



47   */



48  uint8_t flexcan_564xl_init(uint8_t flex_can, uint32_t MCR, uint32_t CR)



49  {



50    uint8_t x = 0U;



51    uint8_t i = 0U;



52    uint8_t status = 0U;



53    switch (flex_can) {



54     case FLEXCAN_0:



55      {



56        CAN_0.MCR.B.MDIS = 0;



57        CAN_0.MCR.B.SOFT_RST = 1;



58        while (CAN_0.MCR.B.SOFT_RST ==1) {



59        }



60



61        CAN_0.MCR.B.MDIS = 1;



62        CAN_0.CR.B.CLK_SRC = 0;



63



64        /*Enable the CAN module */



65        CAN_0.MCR.B.MDIS = 0;



66



67        /* Set FRZ bit */



68        CAN_0.MCR.B.FRZ = 1;



69



70        /* Set HALT bit */



71        CAN_0.MCR.B.HALT = 1;



72        CAN_0.MCR.R = MCR;



73        CAN_0.CR.R = CR;



74



75        /* Set the FlexCAN Maximum Buffer Number */



76        CAN_0.MCR.B.MAXMB = 32;



77



78        /* clear memory from message buffer 0 to 15 */



79        for (x=0; x < 32; x++) {



80          CAN_0.BUF[x].CS.R = 0;



81          CAN_0.BUF[x].ID.R = 0;



82          for (i=0; i < 2; i++) {



83            CAN_0.BUF[x].DATA.W[i] = 0;



84          }



85        }



86



87        for (x=0 ; x<32 ; x++) {



88          CAN_0.RXIMR[x].R = 0xFFFFFFFF;



89        }



90



91        CAN_0.IMRL.R = 0x00000000;



92        CAN_0.MCR.B.HALT = 0;



93        CAN_0.MCR.B.FRZ = 0;



94



95        /* await synchronization (delay) */



96        for (x=1; x < 255; x++) {



97        }



98



99        if (CAN_0.MCR.B.NOT_RDY == 1) {



100          status = 1;



101        } else {



102          status = 0;



103        }



104        break;



105      }



106



107     case FLEXCAN_1:



108      {



109        CAN_1.MCR.B.MDIS = 0;



110        CAN_1.MCR.B.SOFT_RST = 1;



111        while (CAN_1.MCR.B.SOFT_RST ==1) {



112        }



113



114        CAN_1.MCR.B.MDIS = 1;



115        CAN_1.CR.B.CLK_SRC = 0;



116



117        /*Enable the CAN module */



118        CAN_1.MCR.B.MDIS = 0;



119



120        /* Set FRZ bit */



121        CAN_1.MCR.B.FRZ = 1;



122



123        /* Set HALT bit */



124        CAN_1.MCR.B.HALT = 1;



125        CAN_1.MCR.R = MCR;



126        CAN_1.CR.R = CR;



127



128        /* Set the FlexCAN Maximum Buffer Number */



129        CAN_1.MCR.B.MAXMB = 32;



130



131        /* clear memory from message buffer 0 to 15 */



132        for (x=0; x < 32; x++) {



133          CAN_1.BUF[x].CS.R = 0;



134          CAN_1.BUF[x].ID.R = 0;



135          for (i=0; i < 2; i++) {



136            CAN_1.BUF[x].DATA.W[i] = 0;



137          }



138        }



139



140        for (x=0 ; x<32 ; x++) {



141          CAN_1.RXIMR[x].R = 0xFFFFFFFF;



142        }



143



144        CAN_1.IMRL.R = 0x00000000;



145        CAN_1.MCR.B.HALT = 0;



146        CAN_1.MCR.B.FRZ = 0;



147

149        for (x=1; x < 255; x++) {

432  /******************************************************************************



433   *   Function:flexcan_564xl_buffer_transmit



434     @brief    transmit  FlexCAN module buffer.



435     @details   transmit FlexCAN module buffer



436



437     @param[in]     ptr_FLEXCAN_MsgBufferStructure     pointer to structure



438     @param[in]     BufferNo     buffer



439     @param[in]     flex_can     FlexCAN module



440



441     @return none



442   */



443  void flexcan_564xl_buffer_transmit(const FLEXCAN_MsgBufferStructure



444    *ptr_FLEXCAN_MsgBufferStructure,uint8_t BufferNo,uint8_t flex_can)



445  {



446    uint8_t i;



447    if (flex_can==FLEXCAN_0) {



448      if (ptr_FLEXCAN_MsgBufferStructure->ext) {



449        CAN_0.BUF[BufferNo].CS.B.IDE = 1;/* extended ID */



450        CAN_0.BUF[BufferNo].ID.R = ptr_FLEXCAN_MsgBufferStructure->id;



451      } else {



452        CAN_0.BUF[BufferNo].CS.B.IDE = 0;/* standart ID */



453        CAN_0.BUF[BufferNo].ID.B.STD_ID = (uint16_t)



454          ptr_FLEXCAN_MsgBufferStructure->id;



455      }



456



457      CAN_0.BUF[BufferNo].CS.B.RTR = 0;



458      CAN_0.BUF[BufferNo].CS.B.LENGTH = ptr_FLEXCAN_MsgBufferStructure->length;



459      for (i=0; i < ptr_FLEXCAN_MsgBufferStructure->length; i++) {



460        CAN_0.BUF[BufferNo].DATA.B[i] = 0;



461        CAN_0.BUF[BufferNo].DATA.B[i] = ptr_FLEXCAN_MsgBufferStructure->data[i];



462      }



463



464      CAN_0.BUF[BufferNo].CS.B.SRR = 1;/* Tx frame */



465      CAN_0.BUF[BufferNo].CS.B.CODE |= ptr_FLEXCAN_MsgBufferStructure->code;



466    } else if (flex_can==FLEXCAN_1) {



467      if (ptr_FLEXCAN_MsgBufferStructure->ext) {



468        CAN_1.BUF[BufferNo].CS.B.IDE = 1;/* extended ID */



469        CAN_1.BUF[BufferNo].ID.R = ptr_FLEXCAN_MsgBufferStructure->id;



470      } else {



471        CAN_1.BUF[BufferNo].CS.B.IDE = 0;/* standart ID */



472        CAN_1.BUF[BufferNo].ID.B.STD_ID = (uint16_t)



473          ptr_FLEXCAN_MsgBufferStructure->id;



474      }



475



476      CAN_1.BUF[BufferNo].CS.B.RTR = 0;



477      CAN_1.BUF[BufferNo].CS.B.LENGTH = ptr_FLEXCAN_MsgBufferStructure->length;



478      for (i=0; i < ptr_FLEXCAN_MsgBufferStructure->length; i++) {



479        CAN_1.BUF[BufferNo].DATA.B[i] = 0;



480        CAN_1.BUF[BufferNo].DATA.B[i] = ptr_FLEXCAN_MsgBufferStructure->data[i];



481      }



482



483      CAN_1.BUF[BufferNo].CS.B.SRR = 1;/* Tx frame */



484      CAN_1.BUF[BufferNo].CS.B.CODE |= ptr_FLEXCAN_MsgBufferStructure->code;



485    }



486  }
