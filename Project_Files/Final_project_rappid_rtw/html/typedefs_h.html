27#ifdef __MWERKS__                      /* Freescale CodeWarrior */



28#include <stdint.h>



29



30/* Standard typedefs used by header files, based on ISO C standard */



31typedef volatile int8_t vint8_t;



32typedef volatile uint8_t vuint8_t;



33typedef volatile int16_t vint16_t;



34typedef volatile uint16_t vuint16_t;



35typedef volatile int32_t vint32_t;



36typedef volatile uint32_t vuint32_t;



37



38#else



39#ifdef __ghs__                         /* GreenHills*/



40#include <stdint.h>



41



42/* Standard typedefs used by header files, based on ISO C standard */



43typedef volatile int8_t vint8_t;



44typedef volatile uint8_t vuint8_t;



45typedef volatile int16_t vint16_t;



46typedef volatile uint16_t vuint16_t;



47typedef volatile int32_t vint32_t;



48typedef volatile uint32_t vuint32_t;



49



50#else



51



52/* This is needed for compilers that don't have a stdint.h file */



53typedef signed char int8_t;



54typedef unsigned char uint8_t;



55typedef volatile signed char vint8_t;



56typedef volatile unsigned char vuint8_t;



57typedef signed short int16_t;



58typedef unsigned short uint16_t;



59typedef volatile signed short vint16_t;



60typedef volatile unsigned short vuint16_t;



61typedef signed int int32_t;



62typedef unsigned int uint32_t;



63typedef volatile signed int vint32_t;



64typedef volatile unsigned int vuint32_t;

66#endif                                 /* __ghs__ */
