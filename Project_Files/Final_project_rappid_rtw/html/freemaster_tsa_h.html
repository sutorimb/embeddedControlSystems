29/* current TSA version  */

32/* TSA flags carried in TSA_ENTRY.info (except the first entry in table) */



33#define FMSTR_TSA_INFO_ENTRYTYPE_MASK  0x0003U                   /* flags reserved for TSA_ENTRY use */



34#define FMSTR_TSA_INFO_STRUCT          0x0000U                   /* ENTRYTYPE: structure parent type */



35#define FMSTR_TSA_INFO_RO_VAR          0x0001U                   /* ENTRYTYPE: read-only variable */



36#define FMSTR_TSA_INFO_MEMBER          0x0002U                   /* ENTRYTYPE: structure member */



37#define FMSTR_TSA_INFO_RW_VAR          0x0003U                   /* ENTRYTYPE: read-write variable */



38#define FMSTR_TSA_INFO_VAR_FLAG        0x0001U                   /* ENTRYTYPE: FLAG: any variable */



39#define FMSTR_TSA_INFO_RWV_FLAG        0x0002U                   /* ENTRYTYPE: FLAG: R/W access */

41/* TSA table index and size (both unsigned, at least 16 bit wide) */



42typedef FMSTR_SIZE FMSTR_TSA_TINDEX;



43typedef FMSTR_SIZE FMSTR_TSA_TSIZE;



44



45/* pointer types used in TSA tables can be overridden in freemaster.h */



46/* (this is why macros are used instead of typedefs) */



47#ifndef FMSTR_TSATBL_STRPTR



48#define FMSTR_TSATBL_STRPTR            const char*



49#endif



50



51#ifndef FMSTR_TSATBL_STRPTR_CAST



52#define FMSTR_TSATBL_STRPTR_CAST(x)    ((FMSTR_TSATBL_STRPTR)(x))



53#endif



54



55#ifndef FMSTR_TSATBL_VOIDPTR



56#define FMSTR_TSATBL_VOIDPTR           const void*



57#endif



58



59#ifndef FMSTR_TSATBL_VOIDPTR_CAST



60#define FMSTR_TSATBL_VOIDPTR_CAST(x)   ((FMSTR_TSATBL_VOIDPTR)(x))



61#endif



62



63/* TSA table entry. The unions inside assures variables sized enough to */



64/* accommodate both the C-pointer and the user-requested size (FMSTR_ADDR) */



65typedef struct {



66  union {



67    FMSTR_TSATBL_STRPTR p;



68    FMSTR_ADDR n;



69  } name;



70



71  union {



72    FMSTR_TSATBL_STRPTR p;



73    FMSTR_ADDR n;



74  } type;



75



76  union {



77    FMSTR_TSATBL_VOIDPTR p;



78    FMSTR_ADDR n;



79  } addr;



80



81  union {



82    FMSTR_TSATBL_VOIDPTR p;



83    FMSTR_ADDR n;



84  } info;



85} FMSTR_TSA_ENTRY;

87/*//////////////////////////////////////////////// */



88/* single table-building macros */



89#define FMSTR_TSA_FUNC(id)             FMSTR_TsaGetTable_##id



90#define FMSTR_TSA_FUNC_PROTO(id)       const FMSTR_TSA_ENTRY* FMSTR_TSA_FUNC(id) (FMSTR_TSA_TSIZE* pTableSize)



91#define FMSTR_TSA_TABLE_BEGIN(id) \



92 FMSTR_TSA_FUNC_PROTO(id); \



93 FMSTR_TSA_FUNC_PROTO(id) { \



94 static FMSTR_TSA_CDECL FMSTR_TSA_ENTRY fmstr_tsatable[] = {



95



96/* entry info  */



97#define FMSTR_TSA_INFO1(elem, flags)   FMSTR_TSATBL_VOIDPTR_CAST(((sizeof(elem))<<2)|(flags))



98#define FMSTR_TSA_INFO2(size, flags)   FMSTR_TSATBL_VOIDPTR_CAST(((size)<<2)|(flags))



99#define FMSTR_TSA_STRUCT(name) \



100 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(NULL), FMSTR_TSATBL_VOIDPTR_CAST(NULL), FMSTR_TSA_INFO1(name, FMSTR_TSA_INFO_STRUCT) },



101#define FMSTR_TSA_MEMBER(parenttype,name,type) \



102 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(type), FMSTR_TSATBL_VOIDPTR_CAST(&((parenttype*)0)->name), FMSTR_TSA_INFO1(((parenttype*)0)->name, FMSTR_TSA_INFO_MEMBER) },



103#define FMSTR_TSA_RO_VAR(name,type) \



104 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(type), FMSTR_TSATBL_VOIDPTR_CAST(&(name)), FMSTR_TSA_INFO1(name, FMSTR_TSA_INFO_RO_VAR) },



105#define FMSTR_TSA_RW_VAR(name,type) \



106 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(type), FMSTR_TSATBL_VOIDPTR_CAST(&(name)), FMSTR_TSA_INFO1(name, FMSTR_TSA_INFO_RW_VAR) },



107#define FMSTR_TSA_RO_MEM(name,type,addr,size) \



108 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(type), FMSTR_TSATBL_VOIDPTR_CAST(addr), FMSTR_TSA_INFO2(size, FMSTR_TSA_INFO_RO_VAR) },



109#define FMSTR_TSA_RW_MEM(name,type,addr,size) \



110 { FMSTR_TSATBL_STRPTR_CAST(#name), FMSTR_TSATBL_STRPTR_CAST(type), FMSTR_TSATBL_VOIDPTR_CAST(addr), FMSTR_TSA_INFO2(size, FMSTR_TSA_INFO_RW_VAR) },



111#define FMSTR_TSA_TABLE_END()          }; \



112 if(pTableSize) *pTableSize = sizeof(fmstr_tsatable); \



113 return fmstr_tsatable; }

115/*///////////////////////////////////////////////////////////////////// */



116/* TSA "Base Types", all are implemented as a one-char strings */



117/* retrieved by PC and parsed according to the binary scheme */



118/* "111STTZZ" where TT=type[int,frac,fp,x] S=signed ZZ=size[1,2,4,8] */



119#define FMSTR_TSA_UINT8                "\xE0"



120#define FMSTR_TSA_UINT16               "\xE1"



121#define FMSTR_TSA_UINT32               "\xE2"



122#define FMSTR_TSA_UINT64               "\xE3"



123#define FMSTR_TSA_SINT8                "\xF0"



124#define FMSTR_TSA_SINT16               "\xF1"



125#define FMSTR_TSA_SINT32               "\xF2"



126#define FMSTR_TSA_SINT64               "\xF3"



127#define FMSTR_TSA_UFRAC16              "\xE5"



128#define FMSTR_TSA_UFRAC32              "\xE6"



129#define FMSTR_TSA_FRAC16               "\xF5"



130#define FMSTR_TSA_FRAC32               "\xF6"



131#define FMSTR_TSA_FLOAT                "\xFA"



132#define FMSTR_TSA_DOUBLE               "\xFB"



133
