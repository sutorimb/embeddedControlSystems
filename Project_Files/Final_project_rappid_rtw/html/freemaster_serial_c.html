137static void FMSTR_RxQueue(FMSTR_BCHR nRxChar)



138{



139  /* future value of write pointer */



140  FMSTR_BPTR wpnext = pcm_pRQueueWP + 1;



141



142  /*lint -e{946} pointer arithmetic is okay here (same array) */



143  if (wpnext >= (pcm_pRQueueBuffer + FMSTR_COMM_RQUEUE_SIZE)) {



144    wpnext = pcm_pRQueueBuffer;



145  }



146



147  /* any space in queue? */



148  if (wpnext != pcm_pRQueueRP) {



149    *pcm_pRQueueWP = (FMSTR_U8) nRxChar;



150    pcm_pRQueueWP = wpnext;



151  }



152}



153



154#endif                                 /* FMSTR_SHORT_INTR  */

156/*******************************************************************************



157 *



158 * @brief    Late processing of queued characters



159 *



160 * This function takes the queued characters and calls FMSTR_Rx() for each of them,



161 * just like as the characters would be received from SCI one by one.



162 *



163 *******************************************************************************/



164#if FMSTR_SHORT_INTR



165



166static void FMSTR_RxDequeue(void)



167{



168  FMSTR_BCHR nChar = 0U;



169



170  /* get all queued characters */



171  while (pcm_pRQueueRP != pcm_pRQueueWP) {



172    nChar = *pcm_pRQueueRP++;



173



174    /*lint -e{946} pointer arithmetic is okay here (same array) */



175    if (pcm_pRQueueRP >= (pcm_pRQueueBuffer + FMSTR_COMM_RQUEUE_SIZE)) {



176      pcm_pRQueueRP = pcm_pRQueueBuffer;



177    }



178



179    /* emulate the SCI receive event */



180    if (!pcm_wFlags.flg.bTxActive) {



181      (void)FMSTR_Rx(nChar);



182    }



183  }



184}

191/**************************************************************************/



192/*!



193 *



194 * @brief    Handle SCI communication (both TX and RX)



195 *



196 * This function checks the SCI flags and calls the Rx and/or Tx functions



197 *



198 * @note This function can be called either from SCI ISR or from the polling routine



199 *



200 ******************************************************************************/



201void FMSTR_ProcessSCI(void)



202{



203  /* read & clear status     */



204  FMSTR_SCISR wSciSR = FMSTR_SCI_RDCLRSR();



205



206  /* transmitter active and empty? */



207  if (pcm_wFlags.flg.bTxActive) {



208    /* able to accept another character? */



209    if (wSciSR & FMSTR_SCISR_TDRE) {



210      FMSTR_U8 ch;



211



212      //(void)FMSTR_Tx(&ch);



213      /* just put the byte into the SCI transmit buffer */



214      if (!FMSTR_Tx(&ch))



215        FMSTR_SCI_PUTCHAR((FMSTR_U8) ch);



216    }



217



218#if FMSTR_SCI_HAS_TXQUEUE



219



220    /* waiting for transmission complete flag? */



221    if (pcm_wFlags.flg.bTxWaitTC && (wSciSR & FMSTR_SCISR_TC)) {



222      /* after TC, we can switch to listen mode safely */



223      FMSTR_Listen();



224    }



225



226#endif



227



228#if !FMSTR_SCI_TWOWIRE_ONLY



229



230    /* read-out and ignore any received character (loopback) */



231    if (wSciSR & FMSTR_SCISR_RDRF) {



232      /*lint -esym(550, nRxChar) */



233      volatile FMSTR_U16 nRxChar;



234      nRxChar = FMSTR_SCI_GETCHAR();



235    }



236



237#endif



238



239  }



240



241  /* transmitter not active, able to receive */



242  else {



243    /* data byte received? */



244    if (wSciSR & FMSTR_SCISR_RDRF) {



245      FMSTR_BCHR nRxChar = 0U;



246      nRxChar = (FMSTR_BCHR) FMSTR_SCI_GETCHAR();



247



248#if FMSTR_SHORT_INTR



249



250      FMSTR_RxQueue(nRxChar);



251



252#else



253



254      (void)FMSTR_Rx(nRxChar);



255



256#endif



257



258    }



259  }



260}

269static void FMSTR_FlushJtagTx(void);

271/**************************************************************************/



272



273/*!



274 *



275 * @brief    Flush one communication JTAG word



276 *



277 * This function gets the 4 bytes from FreeMASTER TX buffer and place them to



278 * JTAG transmit register.



279 *



280 ******************************************************************************/



281static void FMSTR_FlushJtagTx(void)



282{



283  FMSTR_U8 i;



284  FMSTR_U32 wJtagTxData;



285  register FMSTR_U16 wJtagSR = FMSTR_JTAG_GETSR();



286



287  /* complete word to send */



288  for (i=0; i<4; i++) {



289    FMSTR_U8 ch;



290    if (FMSTR_Tx(&ch))



291      ch = 0xff;



292    wJtagTxData = (wJtagTxData << 8) | ch;



293  }



294



295  /* send the word just completed */



296  FMSTR_JTAG_PUTDWORD(wJtagTxData);



297}

299/**************************************************************************/



300



301/*!



302 *



303 * @brief    Handle JTAG communication (both TX and RX)



304 *



305 * This function checks the JTAG flags and calls the Rx and/or Tx functions



306 *



307 * @note This function can be called either from JTAG ISR or from the polling routine



308 *



309 ******************************************************************************/



310void FMSTR_ProcessJTAG(void)



311{



312  /* read & clear status     */



313  register FMSTR_U16 wJtagSR = FMSTR_JTAG_GETSR();



314



315  /* transmitter active? */



316  if (pcm_wFlags.flg.bTxActive) {



317    /* able to transmit a new character? (TX must be empty = read-out by PC) */



318    if (!(wJtagSR & FMSTR_JTAG_OTXRXSR_TDF)) {



319



320#if FMSTR_USE_JTAG_TXFIX



321



322      /* if TDF bit is useless due to silicon bug, use the RX flag */



323      /* instead (PC sends us a dummy word to kick the RX flag on) */



324      if (wJtagSR & FMSTR_JTAG_OTXRXSR_RDF)



325#endif



326



327        {



328          FMSTR_FlushJtagTx();



329        }



330    }



331



332    /* ignore (read-out) the JTAG-received word */



333    if (wJtagSR & FMSTR_JTAG_OTXRXSR_RDF) {



334      /*lint -esym(550, nRxWord) */



335      volatile FMSTR_U16 nRxWord;



336      nRxWord = FMSTR_JTAG_GETWORD();



337    }



338  }



339



340  /* transmitter not active */



341  else {



342    /* JTAG 32bit word (four bytes) received? */



343    if (wJtagSR & FMSTR_JTAG_OTXRXSR_RDF) {



344      register FMSTR_U32 nRxDWord;



345      FMSTR_INDEX i;



346      nRxDWord = FMSTR_JTAG_GETDWORD();



347



348      /* process all bytes, MSB first */



349      for (i=0; i<4; i++) {



350



351#if FMSTR_SHORT_INTR



352



353        FMSTR_RxQueue((FMSTR_BCHR)((nRxDWord >> 24U) & 0xffU));



354



355#else



356



357        (void)FMSTR_Rx((FMSTR_BCHR)((nRxDWord >> 24U) & 0xffU));



358



359        /* ignore the rest if previous bytes triggered a transmission */



360        /* (i.e. the packet was complete and only filled-up to 32bit word) */



361        if (pcm_wFlags.flg.bTxActive) {



362          break;



363        }



364



365#endif



366



367        /* next byte of 32bit word */



368        nRxDWord = nRxDWord << 8;



369      }



370    }



371  }



372}



373



374#endif                                 /* FMSTR_USE_JTAG */

412/*******************************************************************************



413 *



414 * @brief    Handle MQX IO serial communication (both TX and RX)



415 *



416 * This function calls MQX IO fread() function to get character and process it by



417 *



418 * FMSTR_Rx function when FreeMASTER packet is receiving. This function also transmit



419 *



420 * FreeMASTER response. Character to be send is provided by call of FMSTR_Tx function



421 *



422 * and passed down to fwrite() function.



423 *



424 *******************************************************************************/



425static void FMSTR_ProcessMQXIO(void)



426{



427  if (devfd != NULL) {



428    static FMSTR_U8 TxChar = 0;



429



430    /* transmitter not active, able to receive */



431    if ((!pcm_wFlags.flg.bTxActive) && (!pcm_wFlags.flg.bMqxReadyToSend)) {



432      FMSTR_U8 nRxChar;



433



434      /* read all available bytes from communication interface */



435      while (fread(&nRxChar, 1, 1, devfd)) {



436        if (FMSTR_Rx(nRxChar))



437          break;



438      }



439    }



440



441    /* transmitter active and empty? */



442    if (pcm_wFlags.flg.bTxActive || pcm_wFlags.flg.bMqxReadyToSend) {



443      while (1) {



444        /* write character when is valid */



445        if (pcm_wFlags.flg.bMqxReadyToSend) {



446          if (fwrite(&TxChar, 1, 1, devfd))/* character was successfully send, ready to get next character */



447            pcm_wFlags.flg.bMqxReadyToSend = 0;



448          else                         /* character write failed, needs to be send next call */



449            break;



450        }



451



452        /* is ready to get next character? */



453        if (FMSTR_Tx((FMSTR_U8*)&TxChar))/* FreeMASTER packet is sent, exit loop */



454          break;



455



456        /* read next character, set its validity to be send */



457        pcm_wFlags.flg.bMqxReadyToSend = 1;



458      }



459    }



460  }



461}

548/*!



549 *



550 * @brief    Get data from FreeMASTER TX buffer and send the data to USB device



551 *



552 * This function checks the FreeMASTER transmit state and when FreeMASTER is



553 * ready to send response to PC, copy FreeMASTER TX data to local buffer and

557static void FMSTR_SendUsbPacket(void)



558{



559  /* transmitter active and empty? */



560  if (pcm_wFlags.flg.bTxActive) {



561    FMSTR_U8 index;



562



563    /* fill in the transmitter buffer */



564    for (index = 0; index < DIC_BULK_IN_ENDP_PACKET_SIZE; index++) {



565      FMSTR_U8 TxChar;



566      if (FMSTR_Tx(&TxChar))



567        break;



568      send_buf[index] = TxChar;

570



571    USB_Class_CDC_Interface_DIC_Send_Data(FMSTR_USB_CDC_ID, send_buf,index);



572  }

575/******************************************************************************



576 *



577 *    @name        FMSTR_USB_CDC_Callback



578 *



579 *    @brief       This function handles Class callback



580 *



581 *    @param       controller_ID    : Controller ID



582 *    @param       event_type       : Value of the event



583 *    @param       val              : gives the configuration value



584 *



585 *    @return      None



586 *



587 *****************************************************************************



588 * This function is called from the class layer whenever reset occurs or enum



589 * is complete. After the enum is complete this function sets a variable so



590 * that the application can start.



591 * This function also receives DATA Send and RECEIVED Events



592 *****************************************************************************/



593static void FMSTR_USB_CDC_Callback (



594  FMSTR_U8 controller_ID,              /* [IN] Controller ID */



595  FMSTR_U8 event_type,                 /* [IN] value of the event */



596  void* val                            /* [IN] gives the configuration value */



597  )



598{



599  UNUSED (controller_ID)



600    UNUSED (val)



601    if(event_type == USB_APP_BUS_RESET)



602  {



603    pcm_wFlags.flg.bUsbCdcStartApp = 0U;



604  }



605



606  else if(event_type == USB_APP_ENUM_COMPLETE)



607  {



608    pcm_wFlags.flg.bUsbCdcStartApp = 1U;



609  }



610



611  else if((event_type == USB_APP_DATA_RECEIVED) &&



612          (pcm_wFlags.flg.bUsbCdcStartTrans))



613  {



614    /* manage received data */



615    if (pcm_wFlags.flg.bUsbCdcStartApp) {



616      /* Copy Received Data buffer to Application Buffer */



617      USB_PACKET_SIZE BytesToBeCopied;



618      APP_DATA_STRUCT* dp_rcv = (APP_DATA_STRUCT*)val;



619      BytesToBeCopied = dp_rcv->data_size;



620



621      /* transmitter not active, able to receive */



622      if (!pcm_wFlags.flg.bTxActive) {



623        FMSTR_U8 index;



624        for (index = 0 ; index<BytesToBeCopied ; index++) {



625          if (FMSTR_Rx(dp_rcv->data_ptr[index])) {



626            break;



627          }



628        }



629



630        /* continue receiving, when packet is not decoded */



631        if (index == BytesToBeCopied)  /* receive next packet only when are not received all bytes of packet */



632          (void)USB_Class_CDC_Interface_DIC_Recv_Data(FMSTR_USB_CDC_ID, NULL, 0);



633      }



634    }



635  }



636



637  else if((event_type == USB_APP_SEND_COMPLETE) &&



638          (pcm_wFlags.flg.bUsbCdcStartTrans))



639  {



640    if ((pcm_wFlags.flg.bUsbCdcStartApp) && (pcm_wFlags.flg.bTxActive))/* Finalize USB packet to transmit and send packet */



641      FMSTR_SendUsbPacket();



642    else                               /* Previous Send is complete. Queue next receive */



643      (void)USB_Class_CDC_Interface_DIC_Recv_Data(FMSTR_USB_CDC_ID, NULL, 0);



644  }



645



646  return;



647}

693/*!



694 *



695 * @brief    Start listening on a serial line



696 *



697 * Reset the receiver machine and start listening on a serial line

700static void FMSTR_Listen(void)



701{



702  pcm_nRxTodo = 0U;



703



704  /* disable transmitter state machine */



705  pcm_wFlags.flg.bTxActive = 0U;



706



707#if FMSTR_USE_SCI



708



709  pcm_wFlags.flg.bTxWaitTC = 0U;



710



711  /* disable transmitter, enable receiver (enables single-wire connection) */



712#if !FMSTR_SCI_TWOWIRE_ONLY



713



714  FMSTR_SCI_TD();



715  FMSTR_SCI_RE();



716



717#endif



718#endif                                 /* FMSTR_USE_SCI */



719



720  /* disable transmit, enable receive interrupts */



721#if FMSTR_SHORT_INTR || FMSTR_LONG_INTR



722#if FMSTR_USE_SCI



723



724  FMSTR_SCI_DTXI();                    /* disable SCI transmit interrupt */



725  FMSTR_SCI_ERXI();                    /* enable SCI receive interrupt */



726



727#elif FMSTR_USE_JTAG



728



729  FMSTR_JTAG_DTXI();                   /* disable JTAG transmit interrupt  */



730  FMSTR_JTAG_ERXI();                   /* enable JTAG receive interrupt  */



731



732  /* RIE bit is forced low by HW until EONCE is first accessed, we will try again in FMSTR_Poll */



733  if (!FMSTR_JTAG_ERXI_CHECK())



734    pcm_wFlags.flg.bJtagRIEPending = 1;



735



736#endif                                 /* FMSTR_USE_SCI / FMSTR_USE_JTAG  */



737#endif                                 /* FMSTR_SHORT_INTR || FMSTR_LONG_INTR */

757/**************************************************************************/



758



759/*!



760 *



761 * @brief    Finalize transmit buffer before transmitting



762 *



763 * @param    nLength - response length (1 for status + data length)



764 *



765 *



766 * This Function takes the data already prepared in the transmit buffer



767 * (inlcuding the status byte). It computes the check sum and kicks on tx.



768 *



769 ******************************************************************************/



770void FMSTR_SendResponse(FMSTR_BPTR pResponse, FMSTR_SIZE8 nLength)



771{



772  FMSTR_U16 chSum = 0U;



773  FMSTR_U8 i, c;



774



775  /* remember the buffer to be sent */



776  pcm_pTxBuff = pResponse;



777



778  /* status byte and data are already there, compute checksum only     */



779  for (i=0U; i<nLength; i++) {



780    c = 0U;



781    pResponse = FMSTR_ValueFromBuffer8(&c, pResponse);



782



783    /* add character to checksum */



784    chSum += c;



785



786    /* prevent saturation to happen on DSP platforms */



787    chSum &= 0xffU;



788  }



789



790  /* store checksum after the message */



791  pResponse = FMSTR_ValueToBuffer8(pResponse, (FMSTR_U8) (((FMSTR_U16)~(chSum))



792    + 1U));



793



794  /* send the message and the checksum and the SOB */



795  pcm_nTxTodo = (FMSTR_SIZE8) (nLength + 1U);



796



797  /* now transmitting the response */



798  pcm_wFlags.flg.bTxActive = 1U;



799



800#if FMSTR_USE_USB_CDC || FMSTR_USE_MQX_IO || FMSTR_USE_JTAG



801



802  pcm_wFlags.flg.bTxFirstSobSend = 1U;



803



804#endif



805



806#if FMSTR_USE_SCI



807



808  pcm_wFlags.flg.bTxWaitTC = 0U;



809



810#endif



811



812  /* do not replicate the initial SOB  */



813  pcm_wFlags.flg.bTxLastCharSOB = 0U;



814



815#if FMSTR_USE_SCI



816



817  {



818    /*lint -esym(550, dummySR) */



819    volatile FMSTR_SCISR dummySR;



820



821    /* disable receiver, enable transmitter (single-wire communication) */



822#if !FMSTR_SCI_TWOWIRE_ONLY



823



824    FMSTR_SCI_RD();



825    FMSTR_SCI_TE();



826



827#endif



828



829    /* kick on the SCI transmission (also clears TX Empty flag on some platforms) */



830    dummySR = FMSTR_SCI_GETSR();



831    FMSTR_SCI_PUTCHAR(FMSTR_SOB);



832  }



833



834#elif FMSTR_USE_JTAG



835



836  /* kick on the JTAG transmission */



837  FMSTR_FlushJtagTx();



838



839#endif



840



841  /* TX interrupt enable, RX interrupt disable */



842#if FMSTR_LONG_INTR || FMSTR_SHORT_INTR



843#if FMSTR_USE_SCI



844



845  FMSTR_SCI_DRXI();



846  FMSTR_SCI_ETXI();



847



848#elif FMSTR_USE_JTAG



849#if FMSTR_USE_JTAG_TXFIX



850



851  /* in TX-bugfix mode, keep the RX interrupt enabled as it */



852  /* is used as "able-to-TX" notification from the PC */



853  FMSTR_JTAG_ERXI();



854



855#else



856



857  /* otherwise, JTAG is very same as the SCI */



858  if (pcm_wFlags.flg.bTxActive)        //enable TX interrupt only when is not one word response



859  {



860    FMSTR_JTAG_DRXI();



861    FMSTR_JTAG_ETXI();



862  }



863



864#endif



865



866#elif FMSTR_USE_USB_CDC



867



868  if ((pcm_wFlags.flg.bUsbCdcStartApp) && (pcm_wFlags.flg.bTxActive) &&



869      (pcm_wFlags.flg.bUsbCdcStartTrans))/* Finalize USB packet to transmit and send packet */



870    FMSTR_SendUsbPacket();



871  else                                 /* Previous Send is complete. Queue next receive */



872    (void)USB_Class_CDC_Interface_DIC_Recv_Data(FMSTR_USB_CDC_ID, NULL, 0);



873



874#endif



875#endif                                 /* FMSTR_LONG_INTR || FMSTR_SHORT_INTR */



876



877}

879/**************************************************************************/



880



881/*!



882 *



883 * @brief    Output buffer transmission



884 *



885 * @param  pTxChar  The character to be transmit



886 *



887 * get ready buffer(prepare data to send)



888 *



889 ******************************************************************************/



890FMSTR_BOOL FMSTR_Tx(FMSTR_U8* pTxChar)



891{



892



893#if FMSTR_USE_USB_CDC || FMSTR_USE_MQX_IO || FMSTR_USE_JTAG



894



895  /* to send first SOB byte*/



896  if (pcm_wFlags.flg.bTxFirstSobSend) {



897    *pTxChar = FMSTR_SOB;



898    pcm_wFlags.flg.bTxFirstSobSend = 0U;



899    return FMSTR_FALSE;



900  }



901



902#endif



903



904  if (pcm_nTxTodo) {



905    /* fetch & send character ready to transmit */



906    /*lint -e{534} ignoring return value */



907    FMSTR_ValueFromBuffer8(pTxChar, pcm_pTxBuff);



908



909    /* first, handle the replicated SOB characters */



910    if (*pTxChar == FMSTR_SOB) {



911      pcm_wFlags.flg.bTxLastCharSOB ^= 1U;



912      if ((pcm_wFlags.flg.bTxLastCharSOB)) {



913        /* yes, repeat the SOB next time */



914        return FMSTR_FALSE;



915      }



916    }



917



918    /* no, advance tx buffer pointer */



919    pcm_nTxTodo--;



920    pcm_pTxBuff = FMSTR_SkipInBuffer(pcm_pTxBuff, 1U);



921    return FMSTR_FALSE;



922  }



923



924  /* when SCI TX buffering is enabled, we must first wait until all



925     characters are physically transmitted (before disabling transmitter) */



926#if FMSTR_USE_SCI



927#if FMSTR_SCI_HAS_TXQUEUE



928



929  pcm_wFlags.flg.bTxWaitTC = 1;



930



931  /* wait for SCI TC interrupt */



932#if FMSTR_SHORT_INTR || FMSTR_LONG_INTR



933



934  FMSTR_SCI_ETCI();



935



936#endif



937



938#else



939



940  /* start listening immediately */



941  FMSTR_Listen();



942



943#endif



944



945#else



946



947  /* start listening immediately */



948  FMSTR_Listen();



949



950#endif



951



952  return FMSTR_TRUE;



953}

955/**************************************************************************/



956



957/*!



958 *



959 * @brief  Handle received character



960 *



961 * @param  nRxChar  The character to be processed



962 *



963 * Handle the character received and -if the message is complete- call the



964 * protocol decode routine.



965 *



966 ******************************************************************************/



967FMSTR_BOOL FMSTR_Rx(FMSTR_BCHR nRxChar)



968{



969  FMSTR_SERIAL_FLAGS * pflg = &pcm_wFlags;



970



971  /* first, handle the replicated SOB characters */



972  if (nRxChar == FMSTR_SOB) {



973    pflg->flg.bRxLastCharSOB ^= 1;



974    if (pflg->flg.bRxLastCharSOB) {



975      /* this is either the first byte of replicated SOB or a  */



976      /* real Start-of-Block mark - we will decide next time in FMSTR_Rx */



977      return FMSTR_FALSE;



978    }



979  }



980



981  /* we have got a common character preceded by the SOB -  */



982  /* this is the command code! */



983  if (pflg->flg.bRxLastCharSOB) {



984    /* reset receiving process */



985    pcm_pRxBuff = pcm_pCommBuffer;



986    *pcm_pRxBuff++ = nRxChar;



987



988    /* start computing the checksum */



989    pcm_nRxCheckSum = nRxChar;



990    pcm_nRxTodo = 0U;



991



992    /* if the standard command was received, the message length will come in next byte */



993    pflg->flg.bRxMsgLengthNext = 1U;



994



995    /* fast command? */



996    if (!((~nRxChar) & FMSTR_FASTCMD)) {



997      /* fast command received, there will be no length information */



998      pflg->flg.bRxMsgLengthNext = 0U;



999



1000      /* as it is encoded in the command byte directly */



1001      pcm_nRxTodo = (FMSTR_SIZE8)



1002        (((((FMSTR_SIZE8)nRxChar) & FMSTR_FASTCMD_DATALEN_MASK) >>



1003          FMSTR_FASTCMD_DATALEN_SHIFT) + 1U);



1004    }



1005



1006    /* command code stored & processed */



1007    pflg->flg.bRxLastCharSOB = 0U;



1008    return FMSTR_FALSE;



1009  }



1010



1011  /* we are waiting for the length byte */



1012  if (pflg->flg.bRxMsgLengthNext) {



1013    /* this byte, total data length and the checksum */



1014    pcm_nRxTodo = (FMSTR_SIZE8) (1U + ((FMSTR_SIZE8)nRxChar) + 1U);



1015



1016    /* now read the data bytes */



1017    pflg->flg.bRxMsgLengthNext = 0U;



1018  }



1019



1020  /* waiting for a data byte? */



1021  if (pcm_nRxTodo) {



1022    /* add this byte to checksum */



1023    pcm_nRxCheckSum += nRxChar;



1024



1025    /* decrease number of expected bytes */



1026    pcm_nRxTodo--;



1027



1028    /* was it the last byte of the message (checksum)? */



1029    if (!pcm_nRxTodo) {



1030      /* receive buffer overflow? */



1031      if (pcm_pRxBuff == NULL) {



1032        FMSTR_SendError(FMSTR_STC_CMDTOOLONG);



1033      }



1034



1035      /* checksum error? */



1036      else if ((pcm_nRxCheckSum & 0xffU) != 0U) {



1037        FMSTR_SendError(FMSTR_STC_CMDCSERR);



1038      }



1039



1040      /* message is okay */



1041      else {



1042



1043#if FMSTR_USE_USB_CDC && FMSTR_SHORT_INTR



1044



1045        /* Decode protocol and send response in Poll function */



1046        pflg->flg.bUsbReadyToDecode = 1U;



1047



1048#else



1049



1050        /* do decode now! */



1051        FMSTR_ProtocolDecoder(pcm_pCommBuffer);



1052



1053#endif



1054



1055      }



1056



1057      return FMSTR_TRUE;



1058    }



1059



1060    /* not the last character yet */



1061    else {



1062      /* is there still a space in the buffer? */



1063      if (pcm_pRxBuff) {



1064        /*lint -e{946} pointer arithmetic is okay here (same array) */



1065        if (pcm_pRxBuff < (pcm_pCommBuffer + FMSTR_COMM_BUFFER_SIZE)) {



1066          /* store byte  */



1067          *pcm_pRxBuff++ = nRxChar;



1068        }



1069



1070        /* buffer is full! */



1071        else {



1072          /* NULL rx pointer means buffer overflow - but we still need */



1073          /* to receive all message characters (for the single-wire mode) */



1074          /* so keep "receiving" - but throw away all characters from now */



1075          pcm_pRxBuff = NULL;



1076        }



1077      }



1078    }



1079  }



1080



1081  return FMSTR_FALSE;



1082}
