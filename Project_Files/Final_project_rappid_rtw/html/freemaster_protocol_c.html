109/*!



110 *



111 * @brief    Decodes the FreeMASTER protocol and calls appropriate handlers



112 *



113 * @param    pMessageIO - message in/out buffer



114 *



115 * @return   TRUE if frame was valid and any output was generated to IO buffer



116 *



117 * This Function decodes given message and invokes proper command handler



118 * which fills in the response. The response transmission is initiated



119 * in this call as well.



120 *



121 ******************************************************************************/



122FMSTR_BOOL FMSTR_ProtocolDecoder(FMSTR_BPTR pMessageIO)



123{



124  FMSTR_BPTR pResponseEnd;



125  FMSTR_U8 nCmd;



126



127  /* no EX access by default */



128  FMSTR_SetExAddr(FMSTR_FALSE);



129



130  /* command code comes first in the message */



131  /*lint -e{534} return value is not used */



132  FMSTR_ValueFromBuffer8(&nCmd, pMessageIO);



133



134  /* process command   */



135  switch (nCmd)



136  {



137



138#if FMSTR_USE_READVAR



139



140    /* read byte */



141#if FMSTR_USE_EX_CMDS



142



143   case FMSTR_CMD_READVAR8_EX:



144    FMSTR_SetExAddr(FMSTR_TRUE);



145



146#endif



147



148#if FMSTR_USE_NOEX_CMDS



149



150    /*lint -fallthrough */



151   case FMSTR_CMD_READVAR8:



152



153#endif



154



155    pResponseEnd = FMSTR_ReadVar(pMessageIO, 1U);



156    break;



157



158    /* read word */



159#if FMSTR_USE_EX_CMDS



160



161   case FMSTR_CMD_READVAR16_EX:



162    FMSTR_SetExAddr(FMSTR_TRUE);



163



164#endif



165



166#if FMSTR_USE_NOEX_CMDS



167



168    /*lint -fallthrough */



169   case FMSTR_CMD_READVAR16:



170



171#endif



172



173    pResponseEnd = FMSTR_ReadVar(pMessageIO, 2U);



174    break;



175



176    /* read dword */



177#if FMSTR_USE_EX_CMDS



178



179   case FMSTR_CMD_READVAR32_EX:



180    FMSTR_SetExAddr(FMSTR_TRUE);



181



182#endif



183



184#if FMSTR_USE_NOEX_CMDS



185



186    /*lint -fallthrough */



187   case FMSTR_CMD_READVAR32:



188



189#endif



190



191    pResponseEnd = FMSTR_ReadVar(pMessageIO, 4U);



192    break;



193



194#endif                                 /* FMSTR_USE_READVAR */



195



196#if FMSTR_USE_READMEM



197



198    /* read a block of memory */



199#if FMSTR_USE_EX_CMDS



200



201   case FMSTR_CMD_READMEM_EX:



202    FMSTR_SetExAddr(FMSTR_TRUE);



203



204#endif



205



206#if FMSTR_USE_NOEX_CMDS



207



208    /*lint -fallthrough */



209   case FMSTR_CMD_READMEM:



210



211#endif



212



213    pResponseEnd = FMSTR_ReadMem(pMessageIO);



214    break;



215



216#endif                                 /* FMSTR_USE_READMEM */



217



218#if FMSTR_USE_SCOPE



219



220    /* prepare scope variables */



221#if FMSTR_USE_EX_CMDS



222



223   case FMSTR_CMD_SETUPSCOPE_EX:



224    FMSTR_SetExAddr(FMSTR_TRUE);



225



226#endif



227



228#if FMSTR_USE_NOEX_CMDS



229



230    /*lint -fallthrough */



231   case FMSTR_CMD_SETUPSCOPE:



232



233#endif



234



235    pResponseEnd = FMSTR_SetUpScope(pMessageIO);



236    break;



237



238   case FMSTR_CMD_READSCOPE:



239    pResponseEnd = FMSTR_ReadScope(pMessageIO);



240    break;



241



242#endif                                 /* FMSTR_USE_SCOPE */



243



244#if FMSTR_USE_RECORDER



245



246    /* get recorder status */



247   case FMSTR_CMD_GETRECSTS:



248    pResponseEnd = FMSTR_GetRecStatus(pMessageIO);



249    break;



250



251    /* start recorder */



252   case FMSTR_CMD_STARTREC:



253    pResponseEnd = FMSTR_StartRec(pMessageIO);



254    break;



255



256    /* stop recorder */



257   case FMSTR_CMD_STOPREC:



258    pResponseEnd = FMSTR_StopRec(pMessageIO);



259    break;



260



261    /* setup recorder */



262#if FMSTR_USE_EX_CMDS



263



264   case FMSTR_CMD_SETUPREC_EX:



265    FMSTR_SetExAddr(FMSTR_TRUE);



266



267#endif



268



269#if FMSTR_USE_NOEX_CMDS



270



271    /*lint -fallthrough */



272   case FMSTR_CMD_SETUPREC:



273



274#endif



275



276    pResponseEnd = FMSTR_SetUpRec(pMessageIO);



277    break;



278



279    /* get recorder buffer information (force EX instead of non-EX) */



280#if FMSTR_USE_EX_CMDS



281



282   case FMSTR_CMD_GETRECBUFF_EX:



283    FMSTR_SetExAddr(FMSTR_TRUE);



284



285#elif FMSTR_USE_NOEX_CMDS



286



287    /*lint -fallthrough */



288   case FMSTR_CMD_GETRECBUFF:



289



290#endif



291



292    pResponseEnd = FMSTR_GetRecBuff(pMessageIO);



293    break;



294



295#endif                                 /* FMSTR_USE_RECORDER */



296



297#if FMSTR_USE_APPCMD



298



299    /* accept the application command */



300   case FMSTR_CMD_SENDAPPCMD:



301    pResponseEnd = FMSTR_StoreAppCmd(pMessageIO);



302    break;



303



304    /* get the application command status */



305   case FMSTR_CMD_GETAPPCMDSTS:



306    pResponseEnd = FMSTR_GetAppCmdStatus(pMessageIO);



307    break;



308



309    /* get the application command data */



310   case FMSTR_CMD_GETAPPCMDDATA:



311    pResponseEnd = FMSTR_GetAppCmdRespData(pMessageIO);



312    break;



313



314#endif                                 /* FMSTR_USE_APPCMD */



315



316#if FMSTR_USE_WRITEMEM



317



318    /* write a block of memory */



319#if FMSTR_USE_EX_CMDS



320



321   case FMSTR_CMD_WRITEMEM_EX:



322    FMSTR_SetExAddr(FMSTR_TRUE);



323



324#endif



325



326#if FMSTR_USE_NOEX_CMDS



327



328    /*lint -fallthrough */



329   case FMSTR_CMD_WRITEMEM:



330



331#endif



332



333    pResponseEnd = FMSTR_WriteMem(pMessageIO);



334    break;



335



336#endif                                 /* FMSTR_USE_WRITEMEM */



337



338#if FMSTR_USE_WRITEMEMMASK



339



340    /* write block of memory with a bit mask */



341#if FMSTR_USE_EX_CMDS



342



343   case FMSTR_CMD_WRITEMEMMASK_EX:



344    FMSTR_SetExAddr(FMSTR_TRUE);



345



346#endif



347



348#if FMSTR_USE_NOEX_CMDS



349



350    /*lint -fallthrough */



351   case FMSTR_CMD_WRITEMEMMASK:



352



353#endif



354



355    pResponseEnd = FMSTR_WriteMemMask(pMessageIO);



356    break;



357



358#endif                                 /* FMSTR_USE_WRITEMEMMASK */



359



360#if FMSTR_USE_WRITEVAR && FMSTR_USE_NOEX_CMDS



361



362    /* write byte */



363   case FMSTR_CMD_WRITEVAR8:



364    pResponseEnd = FMSTR_WriteVar(pMessageIO, 1U);



365    break;



366



367    /* write word */



368   case FMSTR_CMD_WRITEVAR16:



369    pResponseEnd = FMSTR_WriteVar(pMessageIO, 2U);



370    break;



371



372    /* write dword */



373   case FMSTR_CMD_WRITEVAR32:



374    pResponseEnd = FMSTR_WriteVar(pMessageIO, 4U);



375    break;



376



377#endif                                 /* FMSTR_USE_WRITEVAR && FMSTR_USE_NOEX_CMDS */



378



379#if FMSTR_USE_WRITEVARMASK && FMSTR_USE_NOEX_CMDS



380



381    /* write byte with mask */



382   case FMSTR_CMD_WRITEVAR8MASK:



383    pResponseEnd = FMSTR_WriteVarMask(pMessageIO, 1U);



384    break;



385



386    /* write word with mask */



387   case FMSTR_CMD_WRITEVAR16MASK:



388    pResponseEnd = FMSTR_WriteVarMask(pMessageIO, 2U);



389    break;



390



391#endif                                 /* FMSTR_USE_WRITEVARMASK && FMSTR_USE_NOEX_CMDS */



392



393#if FMSTR_USE_TSA



394



395    /* get TSA table (force EX instead of non-EX) */



396#if FMSTR_USE_EX_CMDS



397



398   case FMSTR_CMD_GETTSAINFO_EX:



399    FMSTR_SetExAddr(FMSTR_TRUE);



400



401#elif FMSTR_USE_NOEX_CMDS



402



403    /*lint -fallthrough */



404   case FMSTR_CMD_GETTSAINFO:



405



406#endif



407



408    pResponseEnd = FMSTR_GetTsaInfo(pMessageIO);



409    break;



410



411#if FMSTR_USE_EX_CMDS



412



413   case FMSTR_CMD_GETSTRLEN_EX:



414    FMSTR_SetExAddr(FMSTR_TRUE);



415



416#endif



417



418#if FMSTR_USE_NOEX_CMDS



419



420    /*lint -fallthrough */



421   case FMSTR_CMD_GETSTRLEN:



422



423#endif



424



425    pResponseEnd = FMSTR_GetStringLen(pMessageIO);



426    break;



427



428#endif                                 /* FMSTR_USE_TSA */



429



430#if FMSTR_USE_BRIEFINFO



431



432    /* retrieve a subset of board information structure */



433   case FMSTR_CMD_GETINFOBRIEF:



434



435#else



436



437    /* retrieve board information structure */



438   case FMSTR_CMD_GETINFO:



439



440#endif



441



442    pResponseEnd = FMSTR_GetBoardInfo(pMessageIO);



443    break;



444



445#if FMSTR_USE_SFIO



446



447   case FMSTR_CMD_SFIOFRAME_0:



448   case FMSTR_CMD_SFIOFRAME_1:



449    pResponseEnd = FMSTR_SfioFrame(pMessageIO);



450    break;



451



452   case FMSTR_CMD_SFIOGETRESP_0:



453   case FMSTR_CMD_SFIOGETRESP_1:



454    pResponseEnd = FMSTR_SfioGetResp(pMessageIO);



455    break;



456



457#endif                                 /* FMSTR_USE_SFIO */



458



459#if FMSTR_USE_PIPES



460



461   case FMSTR_CMD_PIPE:



462    pResponseEnd = FMSTR_PipeFrame(pMessageIO);



463    break;



464



465#endif                                 /* FMSTR_USE_PIPES */



466



467    /* unknown command */



468   default:



469    pResponseEnd = FMSTR_ConstToBuffer8(pMessageIO, FMSTR_STC_INVCMD);



470    break;



471  }



472



473  /* anything to send back? */



474  if (pResponseEnd != pMessageIO) {



475    /*lint -e{946,960} subtracting pointers is appropriate here */



476    FMSTR_SIZE8 nSize = (FMSTR_SIZE8)(pResponseEnd - pMessageIO);



477    FMSTR_SendResponse(pMessageIO, nSize);



478    return FMSTR_TRUE;



479  } else {



480    /* nothing sent out */



481    return FMSTR_FALSE;



482  }



483}

485/**************************************************************************/



486



487/*!



488 *



489 * @brief    Handling GETINFO or GETINFO_BRIEF



490 *



491 * @param    pMessageIO - original command (in) and response buffer (out)



492 *



493 * @return   As all command handlers, the return value should be the buffer



494 *           pointer where the response output finished (except checksum)



495 *



496 ******************************************************************************/



497FMSTR_BPTR FMSTR_GetBoardInfo(FMSTR_BPTR pMessageIO)



498{



499  FMSTR_BPTR pResponse = pMessageIO;



500  FMSTR_U16 wTmp;



501  FMSTR_U8 *pStr;



502  pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);



503  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_PROT_VER));/* protVer */



504  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_CFG_FLAGS));/* cfgFlags */



505  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_CFG_BUS_WIDTH));/* dataBusWdt */



506  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)



507    (FMSTR_GLOB_VERSION_MAJOR));       /* globVerMajor */



508  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)



509    (FMSTR_GLOB_VERSION_MINOR));       /* globVerMinor */



510  pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_COMM_BUFFER_SIZE));/* cmdBuffSize  */



511



512  /* that is all for brief info */



513#if FMSTR_USE_BRIEFINFO



514



515  FMSTR_UNUSED(pStr);



516  FMSTR_UNUSED(wTmp);



517



518#else                                  /* FMSTR_USE_BRIEFINFO */



519#if FMSTR_USE_RECORDER



520



521  /* recorder buffer size is always measured in bytes */



522  wTmp = FMSTR_GetRecBuffSize();



523  wTmp *= FMSTR_CFG_BUS_WIDTH;



524



525  /* send size and timebase    */



526  pResponse = FMSTR_ValueToBuffer16(pResponse, wTmp);



527  pResponse = FMSTR_ConstToBuffer16(pResponse, (FMSTR_U16) FMSTR_REC_TIMEBASE);



528



529#else                                  /* FMSTR_USE_RECORDER */



530



531  FMSTR_UNUSED(wTmp);



532



533  /* recorder info zeroed */



534  pResponse = FMSTR_ConstToBuffer16(pResponse, 0);



535  pResponse = FMSTR_ConstToBuffer16(pResponse, 0);



536



537#endif                                 /* FMSTR_USE_RECORDER */



538



539#if FMSTR_LIGHT_VERSION



540



541  FMSTR_UNUSED(pStr);



542  pResponse = FMSTR_SkipInBuffer(pResponse, (FMSTR_U8)FMSTR_DESCR_SIZE);



543



544#else



545



546  /* description string */



547  pStr = (FMSTR_U8*) FMSTR_IDT_STRING;



548  for (wTmp = 0U; wTmp < (FMSTR_U8)(FMSTR_DESCR_SIZE); wTmp++) {



549    pResponse = FMSTR_ValueToBuffer8(pResponse, *pStr);



550



551    /* terminating zero used to clear the remainder of the buffer */



552    if (*pStr) {



553      pStr ++;



554    }



555  }



556



557#endif                                 /* SEND_IDT_STRING */



558#endif                                 /* FMSTR_USE_BRIEFINFO */



559



560  return pResponse;



561}

563/**************************************************************************/



564



565/*!



566 *



567 * @brief    Handling READMEM and READMEM_EX commands



568 *



569 * @param    pMessageIO - original command (in) and response buffer (out)



570 *



571 * @return   As all command handlers, the return value should be the buffer



572 *           pointer where the response output finished (except checksum)



573 *



574 ******************************************************************************/



575FMSTR_BPTR FMSTR_ReadMem(FMSTR_BPTR pMessageIO)



576{



577  FMSTR_BPTR pResponse = pMessageIO;



578  FMSTR_ADDR nAddr;



579  FMSTR_U8 nSize;



580  pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);



581  pMessageIO = FMSTR_ValueFromBuffer8(&nSize, pMessageIO);



582  pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);



583



584#if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY



585



586  if (!FMSTR_CheckTsaSpace(nAddr, (FMSTR_SIZE8) nSize, FMSTR_FALSE)) {



587    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_EACCESS);



588  }



589



590#endif



591



592  /* check the response will safely fit into comm buffer */



593  if (nSize > (FMSTR_U8)FMSTR_COMM_BUFFER_SIZE) {



594    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_RSPBUFFOVF);



595  }



596



597  /* success  */



598  pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);



599  return FMSTR_CopyToBuffer(pResponse, nAddr, (FMSTR_SIZE8) nSize);



600}

720/**************************************************************************/

722/*!



723 *



724 * @brief    Handling WRITEMEMMASK and WRITEMEMMASK_EX commands



725 *



726 * @param    pMessageIO - original command (in) and response buffer (out)



727 *



728 * @return   As all command handlers, the return value should be the buffer



729 *           pointer where the response output finished (except checksum)



730 *



731 ******************************************************************************/



732FMSTR_BPTR FMSTR_WriteMemMask(FMSTR_BPTR pMessageIO)



733{



734  FMSTR_BPTR pResponse = pMessageIO;



735  FMSTR_ADDR nAddr;



736  FMSTR_U8 nSize,nResponseCode;



737  pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);



738  pMessageIO = FMSTR_ValueFromBuffer8(&nSize, pMessageIO);



739  pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);



740



741#if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY



742



743  if (!FMSTR_CheckTsaSpace(nAddr, (FMSTR_SIZE8)nSize, FMSTR_TRUE)) {



744    nResponseCode = FMSTR_STC_EACCESS;



745    goto FMSTR_WriteMemMask_exit;



746  }



747



748#endif

750#if FMSTR_CFG_BUS_WIDTH > 1U



751

758#endif

762  nResponseCode = FMSTR_STS_OK;



763



764#if (FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY) || (FMSTR_CFG_BUS_WIDTH > 1U)



765



766 FMSTR_WriteMemMask_exit:



767



768#endif



769



770  return FMSTR_ConstToBuffer8(pResponse, nResponseCode);



771}

773/**************************************************************************/

775/*!



776 *



777 * @brief    Handling WRITEVARMASK command



778 *



779 * @param    pMessageIO - original command (in) and response buffer (out)

781 *



782 * @return   As all command handlers, the return value should be the buffer



783 *           pointer where the response output finished (except checksum)

785 ******************************************************************************/



786FMSTR_BPTR FMSTR_WriteVarMask(FMSTR_BPTR pMessageIO, FMSTR_SIZE8 nSize)



787{

789  FMSTR_ADDR nAddr;



790  FMSTR_U8 nResponseCode;



791  pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 1U);



792  pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);



793



794#if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
