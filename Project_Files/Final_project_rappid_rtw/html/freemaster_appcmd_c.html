71/**************************************************************************/



72



73/*!



74 *



75 * @brief    Handling SANDAPPCMD command



76 *



77 * @param    pMessageIO - original command (in) and response buffer (out)



78 *



79 * @return   As all command handlers, the return value should be the buffer



80 *           pointer where the response output finished (except checksum)



81 *



82 ******************************************************************************/



83FMSTR_BPTR FMSTR_StoreAppCmd(FMSTR_BPTR pMessageIO)



84{



85  FMSTR_BPTR pResponse = pMessageIO;



86  FMSTR_U8 nArgsLen;



87  FMSTR_U8 nCode;



88



89  /* the previous command not yet processed */



90  if (pcm_nAppCmd != FMSTR_APPCMDRESULT_NOCMD) {



91    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_SERVBUSY);



92  }



93



94  pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 1U);



95  pMessageIO = FMSTR_ValueFromBuffer8(&nArgsLen, pMessageIO);



96  pMessageIO = FMSTR_ValueFromBuffer8(&nCode, pMessageIO);



97



98  /* args len is datalen minus one */



99  nArgsLen--;



100



101  /* does the application command fit to buffer ? */



102  if (nArgsLen > (FMSTR_SIZE8) FMSTR_APPCMD_BUFF_SIZE) {



103    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_INVBUFF);



104  }



105



106  /* store command data into dedicated buffer */



107  pcm_nAppCmd = nCode;



108  pcm_nAppCmdLen = nArgsLen;



109



110  /* data copy */



111  if (nArgsLen) {



112    FMSTR_ADDR appCmdBuffAddr;



113    FMSTR_ARR2ADDR(appCmdBuffAddr, pcm_pAppCmdBuff);



114



115    /*lint -e{534} ignoring return value */



116    FMSTR_CopyFromBuffer(appCmdBuffAddr, pMessageIO, (FMSTR_SIZE8) nArgsLen);



117  }



118



119  /* mark command as "running" (without any response data) */



120  pcm_nAppCmdResult = FMSTR_APPCMDRESULT_RUNNING;



121  pcm_nAppCmdResultDataLen = 0U;



122  return FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);



123}

125/**************************************************************************/



126



127/*!



128 *



129 * @brief    Handling GETAPPCMDSTS command



130 *



131 * @param    pMessageIO - original command (in) and response buffer (out)



132 *



133 * @return   As all command handlers, the return value should be the buffer



134 *           pointer where the response output finished (except checksum)



135 *



136 * @note The callback-registerred commands are processed at the moment the PC



137 *       tries to get the result for the first time. At this moment, we are



138 *       sure the PC already got the command delivery packet acknowledged.



139 *



140 ******************************************************************************/



141FMSTR_BPTR FMSTR_GetAppCmdStatus(FMSTR_BPTR pMessageIO)



142{



143



144#if FMSTR_MAX_APPCMD_CALLS



145



146  FMSTR_PAPPCMDFUNC pFunc = NULL;



147  FMSTR_INDEX nIndex;



148



149  /* time to execute the command's callback */



150  if ((nIndex = FMSTR_FindAppCmdCallIndex(pcm_nAppCmd)) >= 0) {



151    pFunc = pcm_pAppCmdCallFunc[nIndex];



152  }



153



154  /* valid callback function found? */



155  if (pFunc) {



156    /* do execute callback, return value is app.cmd result code */



157    pcm_nAppCmdResult = pFunc(pcm_nAppCmd, pcm_pAppCmdBuff, pcm_nAppCmdLen);



158



159    /* nothing more to do with this command (i.e. command acknowledged) */



160    pcm_nAppCmd = FMSTR_APPCMDRESULT_NOCMD;



161  }



162



163#endif



164



165  pMessageIO = FMSTR_ConstToBuffer8(pMessageIO, FMSTR_STS_OK);



166  return FMSTR_ValueToBuffer8(pMessageIO, (FMSTR_U8) pcm_nAppCmdResult);



167}

169/**************************************************************************/



170



171/*!



172 *



173 * @brief    Handling GETAPPCMDDATA command



174 *



175 * @param    pMessageIO - original command (in) and response buffer (out)



176 *



177 * @return   As all command handlers, the return value should be the buffer



178 *           pointer where the response output finished (except checksum)



179 *



180 ******************************************************************************/



181FMSTR_BPTR FMSTR_GetAppCmdRespData(FMSTR_BPTR pMessageIO)



182{



183  FMSTR_BPTR pResponse = pMessageIO;



184  FMSTR_U8 nDataLen;



185  FMSTR_U8 nDataOffset;



186



187  /* the previous command not yet processed */



188  if (pcm_nAppCmd != FMSTR_APPCMDRESULT_NOCMD) {



189    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_SERVBUSY);



190  }



191



192  pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 1U);



193  pMessageIO = FMSTR_ValueFromBuffer8(&nDataLen, pMessageIO);



194  pMessageIO = FMSTR_ValueFromBuffer8(&nDataOffset, pMessageIO);



195



196  /* the response would not fit into comm buffer */



197  if (nDataLen > (FMSTR_U16)FMSTR_COMM_BUFFER_SIZE) {



198    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_RSPBUFFOVF);



199  }



200



201  /* the data would be fetched outside the app.cmd response data */



202  if ((((FMSTR_U16)nDataOffset) + nDataLen) > (FMSTR_SIZE8)



203      pcm_nAppCmdResultDataLen) {



204    return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_INVSIZE);



205  }



206



207  pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);



208



209  /* copy to buffer */



210  {



211    FMSTR_ADDR appCmdBuffAddr;



212    FMSTR_ARR2ADDR(appCmdBuffAddr, pcm_pAppCmdBuff);



213    pResponse = FMSTR_CopyToBuffer(pResponse, appCmdBuffAddr, (FMSTR_SIZE8)



214      nDataLen);



215  }



216



217  return pResponse;



218}

222/*!



223 *



224 * @brief    Find index of registerred app.cmd callback



225 *



226 * @param    nAppcmdCode - App. command ID



227 *



228 * @return   Index of function pointer in our local tables



229 *



230 ******************************************************************************/



231static FMSTR_INDEX FMSTR_FindAppCmdCallIndex(FMSTR_APPCMD_CODE nAppcmdCode)



232{



233



234#if FMSTR_MAX_APPCMD_CALLS > 0



235



236  FMSTR_INDEX i;



237  for (i=0; i<FMSTR_MAX_APPCMD_CALLS; i++) {



238    if (pcm_pAppCmdCallId[i] == nAppcmdCode) {



239      return i;



240    }



241  }



242



243#else



244



245  /*lint -esym(528, FMSTR_FindAppCmdCallIndex) this function is



246     not referenced when APPCMD_CALLS are not used */



247  FMSTR_UNUSED(nAppcmdCode);



248



249#endif



250



251  return -1;



252}

256/*!



257 *



258 * @brief    API: Mark the application command is processed by the application



259 *



260 * @param    nResultCode - the result code which is returned to a host



261 *



262 ******************************************************************************/



263void FMSTR_AppCmdAck(FMSTR_APPCMD_RESULT nResultCode)



264{



265  pcm_nAppCmdResult = nResultCode;



266  pcm_nAppCmdResultDataLen = 0U;



267



268  /* waiting for a new command to come */



269  pcm_nAppCmd = FMSTR_APPCMDRESULT_NOCMD;



270}

272/**************************************************************************/



273



274/*!



275 *



276 * @brief    API: Mark the application command is processed by the application



277 *



278 * @param    pResultDataAddr - address of data we want to return to the PC



279 * @param    nResultDataLen - length of return data



280 *



281 ******************************************************************************/



282void FMSTR_AppCmdSetResponseData(FMSTR_ADDR nResultDataAddr, FMSTR_SIZE



283  nResultDataLen)



284{



285  /* any data supplied by user? */



286  if (nResultDataAddr) {



287    /* response data length is trimmed if response data would not fit into buffer */



288    pcm_nAppCmdResultDataLen = (FMSTR_SIZE8) nResultDataLen;



289    if (pcm_nAppCmdResultDataLen > (FMSTR_SIZE8) FMSTR_APPCMD_BUFF_SIZE) {



290      pcm_nAppCmdResultDataLen = (FMSTR_SIZE8) FMSTR_APPCMD_BUFF_SIZE;



291    }



292



293    if (pcm_nAppCmdResultDataLen > 0U) {



294      FMSTR_ADDR appCmdBuffAddr;



295      FMSTR_ARR2ADDR(appCmdBuffAddr, pcm_pAppCmdBuff);



296      FMSTR_CopyMemory(appCmdBuffAddr, nResultDataAddr, pcm_nAppCmdResultDataLen);



297    }



298  } else {



299    /* no data being returned at all (same effect as pure FMSTR_AppCmdAck) */



300    pcm_nAppCmdResultDataLen = 0U;



301  }



302}

306/*!



307 *



308 * @brief    API: Fetch the application command code if one is ready for processing



309 *



310 * @return   A command code stored in the application cmd buffer.



311 *           The return value is FMSTR_APPCMDRESULT_NOCMD if there is no



312 *           new command since the last call to FMSTR_AppCmdAck



313 *



314 ******************************************************************************/



315FMSTR_APPCMD_CODE FMSTR_GetAppCmd(void)



316{



317



318#if FMSTR_MAX_APPCMD_CALLS



319



320  /* the user can never see the callback-registerred commands */



321  if (FMSTR_FindAppCmdCallIndex(pcm_nAppCmd) >= 0) {



322    return FMSTR_APPCMDRESULT_NOCMD;



323  }



324



325#endif



326



327  /* otherwise, return the appcomand pending */



328  return pcm_nAppCmd;



329}

331/**************************************************************************/



332



333/*!



334 *



335 * @brief    API: Get a pointer to application command data



336 *



337 * @param    pDataLen - A pointer to variable which receives the data length



338 *



339 * @return   Pointer to the "application command" data



340 *



341 ******************************************************************************/



342FMSTR_APPCMD_PDATA FMSTR_GetAppCmdData(FMSTR_SIZE* pDataLen)



343{



344  /* no command, no data */



345  if (pcm_nAppCmd == FMSTR_APPCMDRESULT_NOCMD) {



346    return NULL;



347  }



348



349#if FMSTR_MAX_APPCMD_CALLS



350



351  /* the user never sees the callback-registerred commands */



352  if (FMSTR_FindAppCmdCallIndex(pcm_nAppCmd) >= 0) {



353    return NULL;



354  }



355



356#endif



357



358  /* caller want to know the data length */



359  if (pDataLen) {



360    *pDataLen = pcm_nAppCmdLen;



361  }



362



363  /* data are saved in out buffer */



364  return pcm_nAppCmdLen ? pcm_pAppCmdBuff : (FMSTR_APPCMD_PDATA) NULL;



365}

369/*!



370 *



371 * @brief    API: Register or unregister app.cmd callback handler



372 *



373 * @param    nAppCmdCode   - App.command ID



374 * @param    pCallbackFunc - Pointer to handler function (NULL to unregister)



375 *



376 * @return   Non-zero if successfull, zero if maximum callbacks already set



377 *



378 ******************************************************************************/



379FMSTR_BOOL FMSTR_RegisterAppCmdCall(FMSTR_APPCMD_CODE nAppCmdCode,



380  FMSTR_PAPPCMDFUNC pCallbackFunc)



381{



382



383#if FMSTR_MAX_APPCMD_CALLS > 0



384



385  FMSTR_INDEX nIndex;



386



387  /* keep "void" ID as reserved */



388  if (nAppCmdCode == FMSTR_APPCMDRESULT_NOCMD) {



389    return FMSTR_FALSE;



390  }



391



392  /* get index of app.cmd ID (if already set) */



393  nIndex = FMSTR_FindAppCmdCallIndex(nAppCmdCode);



394



395  /* when not found, get a free slot (only if registerring new callback) */



396  if (nIndex < 0 && pCallbackFunc != NULL) {



397    nIndex = FMSTR_FindAppCmdCallIndex(FMSTR_APPCMDRESULT_NOCMD);



398  }



399



400  /* failed? */



401  if (nIndex < 0) {



402    return FMSTR_FALSE;



403  }



404



405  /* register handler */



406  pcm_pAppCmdCallFunc[nIndex] = pCallbackFunc;



407  pcm_pAppCmdCallId[nIndex] = (FMSTR_APPCMD_CODE) (pCallbackFunc ?



408    nAppCmdCode : FMSTR_APPCMDRESULT_NOCMD);



409  return FMSTR_TRUE;



410



411#else



412



413  FMSTR_UNUSED(pCallbackFunc);



414  FMSTR_UNUSED(nAppCmdCode);



415



416  /* app.cmd callback not implemented */



417  return FMSTR_FALSE;



418



419#endif



420



421}
